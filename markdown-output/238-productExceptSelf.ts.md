function productExceptSelf(nums: number[]): number[] {
    const len = nums.length;
    let res = new Array(len).fill(1);

    let prefix = 1;
    for (let i = 0; i < nums.length; i++) {
        res[i] *= prefix;
        prefix *= nums[i];
    }

    prefix = 1;
    for (let i = len - 1; i >= 0; i--) {
        res[i] *= prefix;
        prefix *= nums[i];
    }
    return res;
};

const nums = [1, 2, 3, 4];
console.log(productExceptSelf(nums));
